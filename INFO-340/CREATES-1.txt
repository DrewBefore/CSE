

CREATE Database Football



CREATE TABLE TEAM
(
TeamID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
TeamName varchar(25) not null,
TeamCity varchar(25) not null,
TeamShortName varchar(25) null,
)

CREATE TABLE CONFERENCE
(
ConferenceID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ConferenceName varchar (100) NOT NULL,
ConferenceDescr varchar(100) NOT NULL
)

CREATE TABLE DIVISION
(
DivisionID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ConferenceID INT FOREIGN KEY REFERENCES Conference (ConferenceID) NOT NULL,
DivisionName varchar(40) not null,
DivisionDescr varchar(100) not null
)

CREATE TABLE DIVISION_TEAM
(
DivisionTeamID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
DivisionID INT FOREIGN KEY REFERENCES Division (DivisionID) NOT NULL,
TeamID INT FOREIGN KEY REFERENCES Team (TeamID) NOT NULL,
BeginDate DATE NOT NULL,
EndDate DATE NULL
)
 
CREATE TABLE TEAM_GAME
(
TeamGameID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
TeamID INT FOREIGN KEY REFERENCES Team (TeamID) NOT NULL,
GameID INT FOREIGN KEY REFERENCES Game (GameID) NOT NULL,
RoleID INT FOREIGN KEY REFERENCES Role (RoleID) NOT NULL,
)

CREATE TABLE TEAM_GAME_STAT
(
TeamGameStatID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
TeamGameID INT FOREIGN KEY REFERENCES Team_Game (TeamGameID) NOT NULL,
StatID INT FOREIGN KEY REFERENCES Stat (StatID) NOT NULL
)


CREATE TABLE GAME
(
GameID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
GameDate Date not null,
GameDescr varchar(100) NLL,
GameName varchar(40) not null
)


CREATE TABLE PLAYER
(
PlayerID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PlayerFName varchar(25) not null,
PlayerLname varchar(25) not null,
DOB date,
Jersey INT,
Height INT,
Weight INT,
ClassYear varchar(25) NOT NULL
)

 
CREATE TABLE PLAYER_POSITION
(
PlayerPositionID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PositionID INT FOREIGN KEY REFERENCES Position (PositionID) NOT NULL,
PlayerID INT FOREIGN KEY REFERENCES Player (PlayerID) NOT NULL
)
 
CREATE TABLE POSITION
(
PositionID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PosName varchar(25) not null,
PosAbbrev varchar(25) not null,
PosDescr varchar(100) NULL,
)


CREATE TABLE COLLEGE
(
CollegeID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
CollegeName varchar(40) not null,
CollegeCity varchar(40) not null,
)

CREATE TABLE PLAYER_COLLEGE
(
PlayerCollegeID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PlayerID INT FOREIGN KEY REFERENCES Player (PlayerID) NOT NULL,
CollegeID INT FOREIGN KEY REFERENCES College(CollegeID) NOT NULL
)
 
CREATE TABLE ROSTER
(
RosterID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PlayerID INT FOREIGN KEY REFERENCES Player (PlayerID) NOT NULL,
TeamID INT FOREIGN KEY REFERENCES Team (TeamID) NOT NULL,
RosterBeginDate Date not null,
RosterEndDate Date not null
)
 
CREATE TABLE STAT
(
StatID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
StatName VarChar(25) not null,
StatDescr VarChar(90) NOT NULL
)
 
CREATE TABLE ROSTER_PLAY_STAT
(
RosterPlayStatID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
RosterID INT FOREIGN KEY REFERENCES Roster (RosterID) NOT NULL,
PlayID INT FOREIGN KEY REFERENCES Play (PlayID) NOT NULL,
StatID INT FOREIGN KEY REFERENCES Stat (StatID) NOT NULL,
StatValue Varchar(25) NULL
)

CREATE TABLE PLAY
(
PlayID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
TeamGameID INT FOREIGN KEY REFERENCES Team_Game (TeamGameID) NOT NULL,
Direction Varchar(25) NULL,
Yardline INT NULL,
Quarter INT NULL,
Time DATETIME NULL
)

CREATE TABLE PLAY_TAG
(
PlayTagID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PlayID INT FOREIGN KEY REFERENCES Play (PlayId) NOT NULL,
TagID INT FOREIGN KEY REFERENCES Tag (TagID) NOT NULL,
)

CREATE TABLE TAG
(
TagID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
PlayTypeID INT FOREIGN KEY REFERENCES Tag_Type (TagTypeID) NOT NULL,
)

CREATE TABLE TAG_TYPE
(
TagTypeID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
TagTypeName Varchar(40) NULL,
TagTypeDescr Varchar(100) NULL
)

CREATE TABLE ROLE
(
RoleID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
RoleName Varchar(40) NOT NULL,
RoleDescr varchar(100) NULL,
)
